**Goal:** Make `SellOrder` a true aggregate root that owns its `Uuid` identity from birth, remove infrastructure-level business-logic bypasses from both repositories, and simplify the mapper to a clean upsert flow.

**Architecture:** `SellOrder` auto-generates `Uuid::v7()` in its constructor; `reconstitute()` is kept as the explicit rehydration entry point used by the mapper. `save()` on both repositories becomes an upsert (detect existing by ID → sync all fields or create fresh). Callers work with domain objects directly: mutate via domain method → call `save()`.

```php
<?php

declare(strict_types=1);

namespace App\Infrastructure\Doctrine\Mapper;

use App\Domain\Position\SellOrder;
use App\Infrastructure\Doctrine\Entity\SellOrderDoctrine;

class SellOrderDoctrineMapper
{
    /**
     * Maps a domain SellOrder to a Doctrine entity for persistence.
     * Pass $existing (a tracked entity) for updates; omit for inserts.
     */
    public function toDoctrineEntity(
        SellOrder $order,
        ?SellOrderDoctrine $existing = null,
    ): SellOrderDoctrine {
        $entity = $existing ?? new SellOrderDoctrine($order->getId());

        return $entity
            ->setSymbol($order->getSymbol())
            ->setBuyPrice($order->getBuyPrice())
            ->setCount($order->getCount())
            ->setStopLossPrice($order->getStopLossPrice())
            ->setInvestment($order->getInvestment())
            ->setSellPrice($order->getSellPrice())
            ->setTier($order->getTier())
            ->setEntryTime($order->getEntryTime())
            ->setStatus($order->getStatus())
            ->setClosePrice($order->getClosePrice())
            ->setCloseReason($order->getCloseReason())
            ->setClosedAt($order->getClosedAt());
    }

    /**
     * Reconstitutes a domain SellOrder from a Doctrine entity (rehydration).
     */
    public function toDomainEntity(SellOrderDoctrine $doctrine): SellOrder
    {
        return SellOrder::reconstitute(
            id:            $doctrine->getId(),
            symbol:        $doctrine->getSymbol(),
            buyPrice:      $doctrine->getBuyPrice(),
            count:         $doctrine->getCount(),
            stopLossPrice: $doctrine->getStopLossPrice(),
            investment:    $doctrine->getInvestment(),
            sellPrice:     $doctrine->getSellPrice(),
            tier:          $doctrine->getTier(),
            entryTime:     $doctrine->getEntryTime(),
            status:        $doctrine->getStatus(),
            closePrice:    $doctrine->getClosePrice(),
            closeReason:   $doctrine->getCloseReason(),
            closedAt:      $doctrine->getClosedAt(),
        );
    }
}
```

```php
<?php

declare(strict_types=1);

namespace App\Infrastructure\Doctrine\Repository;

use App\Domain\Position\SellOrder;
use App\Domain\Position\SellOrderRepositoryInterface;
use App\Infrastructure\Doctrine\Entity\SellOrderDoctrine;
use App\Infrastructure\Doctrine\Mapper\SellOrderDoctrineMapper;
use Doctrine\Bundle\DoctrineBundle\Repository\ServiceEntityRepository;
use Doctrine\Persistence\ManagerRegistry;

class SellOrderDoctrineRepository extends ServiceEntityRepository implements SellOrderRepositoryInterface
{
    public function __construct(
        ManagerRegistry $registry,
        private readonly SellOrderDoctrineMapper $mapper,
    ) {
        parent::__construct($registry, SellOrderDoctrine::class);
    }

    public function save(SellOrder $order): void
    {
        $existing = $this->find($order->getId());
        $entity   = $this->mapper->toDoctrineEntity($order, $existing);

        $em = $this->getEntityManager();
        $em->persist($entity);
        $em->flush();
    }

    public function findBy(array $criteria, ?array $orderBy = null, $limit = null, $offset = null): array
    {
        $entities = parent::findBy($criteria, $orderBy, $limit, $offset);

        return array_map(fn($e) => $this->mapper->toDomainEntity($e), $entities);
    }

    public function remove(SellOrder $order): void
    {
        $entity = $this->find($order->getId());
        if ($entity === null) {
            throw new \RuntimeException("SellOrder {$order->getId()} not found");
        }

        $em = $this->getEntityManager();
        $em->remove($entity);
        $em->flush();
    }
}
```

Note: `$this->find($order->getId())` passes a `Uuid` object directly — Doctrine's `find()` accepts the Symfony `Uuid` type natively for `uuid`-typed ID columns.

```php
<?php

declare(strict_types=1);

namespace App\Infrastructure\InMemory\Repository;

use App\Domain\Position\SellOrder;
use App\Domain\Position\SellOrderRepositoryInterface;

class InMemorySellOrderRepository implements SellOrderRepositoryInterface
{
    /** @var array<string, SellOrder> */
    private array $orders = [];

    public function save(SellOrder $order): void
    {
        $this->orders[$order->getId()->toRfc4122()] = $order;
    }

    public function findBy(array $criteria, ?array $orderBy = null, $limit = null, $offset = null): array
    {
        $result = [];
        foreach ($this->orders as $order) {
            if ($order->getSymbol() === $criteria['symbol'] && $order->getStatus() === $criteria['open']) {
                $result[] = $order;
            }
        }

        return $result;
    }

    public function remove(SellOrder $order): void
    {
        unset($this->orders[$order->getId()->toRfc4122()]);
    }

    /** @return SellOrder[] */
    public function findOpenBySymbol(string $symbol): array
    {
        $result = [];
        foreach ($this->orders as $order) {
            if ($order->getSymbol() === $symbol && $order->getStatus() === 'open') {
                $result[] = $order;
            }
        }

        return $result;
    }
}
```

```php
$positionsById = array_combine(
    array_map(fn(SellOrder $o) => $o->getId()->toRfc4122(), $openPositions),
    $openPositions

// 1. Persist trailing stop updates (engine already mutated the domain objects)
foreach ($tick->stopUpdates as $id => $newStop) {

    $this->emit($output, 'STOP_UPDATE', [

$feeMultiplier = $config->get('feePercent') / 100;
foreach ($tick->closures as $id => $closeInfo) {

    $revenue = $closeInfo['price'] * $order->getCount();

    $this->emit($output, 'SELL', [
        'id'       => $id,
        'candleTs' => date('c', $candleTs),
        'price'    => round($closeInfo['price'], 6),
        'buyPrice' => round($order->getBuyPrice(), 6),
        'reason'   => $closeInfo['reason'],
        'tier'     => $order->getTier(),
        'qty'      => $order->getCount(),
        'profit'   => round($profit, 2),
    ]);
```
